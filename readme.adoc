
= hugo

== 新建
[source,shell script]
----
hugo new site quickstart

----

== 添加主题
[source,shell script]
----
cd quickstart

git init
git submodule add https://github.com/budparr/gohugo-theme-ananke.git themes/ananke

# 编辑你的 config.toml 配置文件使用该主题
echo theme="ananke" >> config.toml

----

== 开始
[source,shell script]
----
hugo new post/my-first-post.md

hugo server -D
----
